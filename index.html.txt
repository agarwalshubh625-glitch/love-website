<script>
// Typing effect
const text = "Shrutiiii 🐼... I am really sorry 😔 I love you more than anything ❤️ Please forgive me 🥺";
let i = 0;

function typeEffect() {
  if (i < text.length) {
    document.getElementById("typingText").innerHTML += text.charAt(i);
    i++;
    setTimeout(typeEffect, 50);
  }
}
typeEffect();

// Music
const music = document.getElementById("bgMusic");

// Mobile autoplay fix
document.body.addEventListener("click", function () {
  music.play();
}, { once: true });

function toggleMusic() {
  if (music.paused) {
    music.play();
  } else {
    music.pause();
  }
}

// Hearts messages
const messages = [
  "I love you Shrutiiii ❤️",
  "Sorry baby 🥺",
  "My panda 🐼❤️",
  "Please forgive me 😔",
  "You are my everything 💖"
];

function startLove() {
  document.body.innerHTML = "<h1 style='color:white;margin-top:40%;'>I LOVE YOU SHRUTIIII 🐼❤️🥰</h1>";
  setInterval(createHeart, 300);
}

function createHeart() {
  const heart = document.createElement("div");
  heart.className = "heart";
  heart.innerText = messages[Math.floor(Math.random() * messages.length)];
  heart.style.left = Math.random() * 100 + "vw";
  document.body.appendChild(heart);

  setTimeout(() => heart.remove(), 6000);
}

// Move NO button
function moveNo() {
  let x = Math.random() * (window.innerWidth - 100);
  let y = Math.random() * (window.innerHeight - 50);
  document.getElementById("no").style.left = x + "px";
  document.getElementById("no").style.top = y + "px";
}
